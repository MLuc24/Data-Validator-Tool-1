create extension if not exists pgcrypto;

drop view if exists public.v_upload_row_details cascade;
drop view if exists public.v_cost_center_details cascade;

drop table if exists public.upload_rows cascade;
drop table if exists public.upload_batches cascade;
drop table if exists public.app_users cascade;
drop table if exists public.cost_centers cascade;
drop table if exists public.plans cascade;
drop table if exists public.companies cascade;

drop function if exists public.set_updated_at() cascade;

create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

create table public.companies (
    company_id integer primary key,
    company_name text not null unique,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
);

create trigger trg_companies_updated_at
before update on public.companies
for each row
execute function public.set_updated_at();

create table public.plans (
    plan_id integer primary key,
    plan_name text not null unique,
    parent_name text,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now()
);

create trigger trg_plans_updated_at
before update on public.plans
for each row
execute function public.set_updated_at();

create table public.cost_centers (
    cost_center_code text primary key,
    cost_center_name text not null,
    company_id integer not null references public.companies(company_id) on update cascade on delete restrict,
    plan_id integer not null references public.plans(plan_id) on update cascade on delete restrict,
    is_active boolean not null default true,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now(),
    constraint uq_cost_center_name_company unique (company_id, cost_center_name)
);

create index idx_cost_centers_company_id on public.cost_centers(company_id);
create index idx_cost_centers_plan_id on public.cost_centers(plan_id);

create trigger trg_cost_centers_updated_at
before update on public.cost_centers
for each row
execute function public.set_updated_at();

create table public.app_users (
    user_id uuid primary key default gen_random_uuid(),
    auth_user_id uuid unique,
    full_name text not null,
    email text unique,
    cost_center_code text references public.cost_centers(cost_center_code) on update cascade on delete set null,
    is_active boolean not null default true,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now(),
    constraint chk_app_users_full_name check (length(trim(full_name)) > 0)
);

create index idx_app_users_cost_center_code on public.app_users(cost_center_code);
create index idx_app_users_is_active on public.app_users(is_active);

create trigger trg_app_users_updated_at
before update on public.app_users
for each row
execute function public.set_updated_at();

create table public.upload_batches (
    batch_id uuid primary key default gen_random_uuid(),
    upload_date date not null,
    accountant_name text not null,
    company_id integer not null references public.companies(company_id) on update cascade on delete restrict,
    cost_center_code text not null references public.cost_centers(cost_center_code) on update cascade on delete restrict,
    bp text,
    file_name text not null,
    file_type text,
    total_rows integer not null default 0,
    preview_rows integer not null default 0,
    status text not null default 'submitted',
    uploaded_by uuid references public.app_users(user_id) on update cascade on delete set null,
    note text,
    created_at timestamptz not null default now(),
    updated_at timestamptz not null default now(),
    constraint chk_upload_batches_total_rows check (total_rows >= 0),
    constraint chk_upload_batches_preview_rows check (preview_rows >= 0),
    constraint chk_upload_batches_status check (
        status in ('draft', 'validated', 'submitted', 'failed')
    )
);

create index idx_upload_batches_upload_date on public.upload_batches(upload_date);
create index idx_upload_batches_company_id on public.upload_batches(company_id);
create index idx_upload_batches_cost_center_code on public.upload_batches(cost_center_code);
create index idx_upload_batches_uploaded_by on public.upload_batches(uploaded_by);
create index idx_upload_batches_status on public.upload_batches(status);

create trigger trg_upload_batches_updated_at
before update on public.upload_batches
for each row
execute function public.set_updated_at();

create table public.upload_rows (
    row_id bigserial primary key,
    batch_id uuid not null references public.upload_batches(batch_id) on update cascade on delete cascade,
    row_number integer not null,
    file_date date,
    system_code text,
    metric_group text,
    category_name text,
    subcategory_name text,
    amount numeric(18,2),
    attribute_name text,
    content_text text,
    company_name_in_file text,
    data_type text,
    block_name text,
    department_name text,
    raw_json jsonb,
    created_at timestamptz not null default now(),
    constraint uq_upload_rows_batch_row_number unique (batch_id, row_number),
    constraint chk_upload_rows_row_number check (row_number > 0)
);

create index idx_upload_rows_batch_id on public.upload_rows(batch_id);
create index idx_upload_rows_file_date on public.upload_rows(file_date);
create index idx_upload_rows_metric_group on public.upload_rows(metric_group);
create index idx_upload_rows_category_name on public.upload_rows(category_name);
create index idx_upload_rows_system_code on public.upload_rows(system_code);
create index idx_upload_rows_raw_json on public.upload_rows using gin(raw_json);

create or replace view public.v_cost_center_details as
select
    cc.cost_center_code,
    cc.cost_center_name,
    c.company_id,
    c.company_name,
    p.plan_id,
    p.plan_name,
    p.parent_name,
    cc.is_active,
    cc.created_at,
    cc.updated_at
from public.cost_centers cc
join public.companies c on c.company_id = cc.company_id
join public.plans p on p.plan_id = cc.plan_id;

create or replace view public.v_upload_row_details as
select
    ur.row_id,
    ur.batch_id,
    ub.upload_date,
    ub.accountant_name,
    ub.bp,
    ub.file_name,
    ub.status,
    c.company_id,
    c.company_name as selected_company_name,
    cc.cost_center_code,
    cc.cost_center_name,
    p.plan_id,
    p.plan_name,
    ur.row_number,
    ur.file_date,
    ur.system_code,
    ur.metric_group,
    ur.category_name,
    ur.subcategory_name,
    ur.amount,
    ur.attribute_name,
    ur.content_text,
    ur.company_name_in_file,
    ur.data_type,
    ur.block_name,
    ur.department_name,
    ur.raw_json,
    ur.created_at
from public.upload_rows ur
join public.upload_batches ub on ub.batch_id = ur.batch_id
join public.companies c on c.company_id = ub.company_id
join public.cost_centers cc on cc.cost_center_code = ub.cost_center_code
join public.plans p on p.plan_id = cc.plan_id;
